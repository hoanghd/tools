Object
  - without( array, values|keys )    
    array_diff( $array, $value );
    array_diff_key( $array, array_flip( $keys ) );
    
  - pick( list, properties )
    
Collection
  - where( list, properties )
    array_filter( $list, function( $item ) use ( $properties ) {
        $diff = array_diff_assoc( $properties, $item );
        return empty( $diff ) ? true : false;
    });
    
  - pluck(list, properties) #Lấy một số field của row trong list $properties = [ 'id', 'age' ]
    array_map( function( $item ) use ( $properties ) {
	    return array_intersect_key( $item, array_flip( $properties ) );
    }, $list);
